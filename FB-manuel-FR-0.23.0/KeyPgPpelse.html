#else
 
Directive conditionnelle du préprocesseur

Syntaxe

#if (expression)
' Déclarations incluses à condition qu'expression soit Vraie (True)
#else
' Déclarations incluses à condition qu'expression soit Fausse (False)
#endif

Description

#else peut être ajouté à un bloc #if, #ifdef, ou #ifndef pour procurer un résultat alternatif à l'expression conditionnelle.

Exemple

#define MODULE_VERSION 1
Dim a As String
#if (MODULE_VERSION > 0)
  a = "Release"
#else
  a = "Beta"
#endif
Print "Program is "; a
